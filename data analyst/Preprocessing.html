# Hm x l invalid data.
def fixing(datum):
  if (not isinstance(datum, str) and math.isnan(datum)): # Nu n l nan
    return np.nan
  elif len(str(datum)) > 4: 
    results = [e for e in re.split(r'[^0-9]', datum) if e != ''] # Chui xc nh nhng ci ko phi s.
    if len(results) == 0:
      return 0
    return max(map(int, results))
  return datum

def bound(x):
  q3, q1 = np.quantile(x, 0.75), np.quantile(x, 0.25)
  IQR = q3 - q1
  return [q1 - 1.5*IQR, q3 + 1.5*IQR]

# dict nhng tn cng ngh cn thay th v 1 tn chung. (jS --> javasript, k8s --> kubernetes, ...)
dict_apply_duplicate = {
    r'js': 'javascript',
    r'k8s': 'kubernetes',
    r'kubrrnetes': 'kubernetes',
    r'gcp': 'google cloud platform',
    r'\.net': '.net',
    r'[^\w](\.net)': '.net core',
    'core core': 'core',
    r'frontend:': 'frontend,',
    r'react': 'react.js',
    r'reactjs': 'react.js',
    r'nodejs': 'node.js',
    r'node': 'node.js',
    r'ts': 'typescript',
    r'[\(\)]': ''
}


def doing(data: str):
    deprecated = ['none', '-', '--', 'nothing'] # list nhng ci gi tr b
    # di y l if data l nan hoc data nm trong list nhng ci b th mnh tr v no_tech
    if (not isinstance(data, str) and math.isnan(data)) or (data in deprecated):
        return ['no_tech']
    
    data = data.lower() # lowercase d liu chui
    # gi ci dict  trn
    # thay th key bng value c du , hai u  tin split
    global dict_apply_duplicate
    for regex, sub in dict_apply_duplicate.items(): # Thay th cc chui vit tt thnh tn y  theo value ca dict
        sub_data = ',%s,' % (sub)
        data = re.compile(regex).sub(sub_data, data)
    
    data = data.replace('/', ',') # thay du / thnh ,
    data = data.split(',') # split cng ngh theo ,
    data = [datum.strip() for datum in data] # loi b space hai u ca tng cng ngh
    data = [datum for datum in data if datum != ''] # nu loi b xong l rng th khng ly 
    data = ['_'.join(datum.split()) for datum in data] # ghp li theo _
    
    if data == ['aws_hadoop_postgre_typescript']: # i vi trng hp c th nh th ny th thay th ring
        data = data[0].split('_')
        
    return data

Hm doing: Input l cc gi tr ca ct MainTech thng qua phng thc apply. X l ri tr v gi tr ngay ti im d liu .

  1. Kim tra nu trong list deprecated hoc l nan gn "no_tech".
  2. Lower case c chui v ch thng.
  3. Thay th cc chui vit tt thnh tn y theo value ca dict_apply_duplicate.
  4. Thay du "/" thnh ",".
  5. Tch chui theo du ","
